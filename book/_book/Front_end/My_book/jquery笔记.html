    // let ocheckAll1 = $('#check_all')[0];
    let acheck = $('.id-check');    //需要被全选的按钮acheck们

      ocheckAll.click(function(){
        if(ocheckAll[0].checked){
          for(let i=0; i<acheck.length; i++){
            acheck[i].checked = true;
          }
        }else{
          for(let i=0; i<acheck.length; i++){
            acheck[i].checked = false;
          }
        }
      })
